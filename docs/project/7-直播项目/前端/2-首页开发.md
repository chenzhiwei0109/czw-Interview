## 逻辑添加

```js
openLive(id) {
    uni.navigateTo({
        url: '../live/liveid？id='+id
    });
},
```

```js
data() {
		return {
			list: [],
			page: 1,
			loadText: '上拉加载更多'
		};
},
getData() {
    let page = this.page;
    return this.$H
        .get('/live/list/' + page)
        .then(res => {
        (this.list = page === 1 ? res : [...this.list, ...res]), (this.loadText = res.length < 10 ? '没有更多了' : '上拉加载更多');
    })
        .catch(err => {
        if (this.page > 1) {
            this.page--;
            this.loadText = '上拉加载更多';
        }
    });
}
```

```js
onLoad() {
	this.getData();
},
onPullDownRefresh() {
    this.page = 1;
    this.getData()
        .then(res => {
        uni.showToast({
            title: '刷新成功',
            icon: 'none'
        });
        uni.stopPullDownRefresh();
    })
        .catch(err => {
        uni.stopPullDownRefresh();
    });
},
onReachBottom() {
	if (this.loadText !== '上拉加载更多') {
		return;
	}
	this.loadText = '加载中...';
	this.page++;
	this.getData();
},
```
