## 推流组件设置(通用模板)

- 取消原生导航栏

- 高度为

  ```vue
  <live-pusher :style="'height: '+windowHeight+'px;'" style="width: 750rpx;"></live-pusher>
  <script>
      export default {
          data() {
              return {
                  windowHeight:0
              }
          },
          onLoad() {
              let res = uni.getSystemInfoSync()
              this.windowHeight = res.windowHeight
          },
  </script>
  ```

- 开启预览

  ```js
  this.context = uni.createLivePusherContext('livePusher', this)
  this.context.startPreview({
      success:(e)=>{
      }
  })
  ```

  

```html
<template>
    <view>


        <live-pusher
                     id='livePusher'
                     ref="livePusher" 
                     class="livePusher" 
                     :url="url"
                     :mode="mode"//推流视频模式，可取值：SD（标清）, HD（高清）, FHD（超清）。	
                     :enable-camera="enableCamera" 
                     :auto-focus="true" 
                     :device-position="position"
                     :beauty="beauty" //美白
                     :whiteness="whiteness"
                     aspect="9:16" 
                     @statechange="statechange" 
                     @netstatus="netstatus" 
                     @error = "error"
                     :style="'height: '+windowHeight+'px;'"
                     style="width: 750rpx;"></live-pusher>



    </view>
</template>

<script>
    export default {
        data() {
            return {
                url:"",
                mode:"SD",
                enableCamera:true,
                position:"back",
                beauty:0,
                whiteness:0,
                windowHeight:0,
                context:null
            }
        },
        onLoad() {
            let res = uni.getSystemInfoSync()
            this.windowHeight = res.windowHeight
        },
        onReady() {
            // 拿到组件上下文
            this.context = uni.createLivePusherContext('livePusher', this)
            this.startPreview()
        },
        methods: {
            // 开启预览
            startPreview(){
                this.context.startPreview({
                    success:(e)=>{
                        console.log(e);
                    }
                })
            },
            // 直播状态变化
            statechange(e){
                console.log(e);
            },
            // 直播网络变化
            netstatus(e){
                console.log(e);
            },
            error(e){
                console.log(e);
            }
        }
    }
</script>

<style>

</style>

```

## 切换封面交互

- 创建图片的baseUrl

```js
//	\common\config.js
export default {
	baseUrl: "http://192.168.101.10:7001",
	socketUrl: "http://192.168.101.10:7001",
	imageUrl: "http://192.168.101.10:7001",
}

```

- 切换封面

```js
import $H from "@/common/request.js";
import $C from "@/common/config.js";
export default {
    data(){
        return {
            form: {
                title: "",
                cover: "",
            },
        }
    },
    methods: {
        //选择封面
        chooseCover() {
            uni.chooseImage({
                count: 1,
                success: res => {
                    $H.upload(
                        '/upload',
                        {
                            filePath: res.tempFilePaths[0]
                        },
                        p => {
                            console.log(p);
                        }
                    ).then(res => {
                        this.form.cover = $C.imageUrl + res.url;
                    });
                }
            });
        },
    }
}
```
