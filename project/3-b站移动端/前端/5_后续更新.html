# 创建弹幕

'use strict';

const db = uniCloud.database()

// 集合
const danmuCollection = db.collection('danmu_table')
// const dbCmd = db.command


exports.main = async (event, context) => {
    //event为客户端上传的参数

    let { id } = event;

    let res = await danmuCollection.add({
        "videoid": id,
        "danmuList": [{
            "text": '即将有一大波弹幕来临~',
            "color": "#ff0000",
            "time": 1
        }
                     ]
    })

    //返回数据给客户端
    return res
};

uniCloud.callFunction({
    name: 'createDanmu',
    data: {
        id
    },
    success: res => {
        console.log(res);
    },
    fail: err => {
        console.log(err);
    }
});

# 获取弹幕

'use strict';

const db = uniCloud.database()

// 集合
const danmuCollection = db.collection('danmu_table')
exports.main = async (event, context) => {
    //event为客户端上传的参数
    let {videoid} = event
    let res = await danmuCollection.where({
        videoid
    }).get()

    //返回数据给客户端
    return res
};

# 发送弹幕

'use strict';


const db = uniCloud.database()

// 集合
const danmuCollection = db.collection('danmu_table')
const dbCmd = db.command


exports.main = async (event, context) => {
    //event为客户端上传的参数
    let {videoid,danmuItem} = event

    let res = await danmuCollection.where({
        videoid
    }).update({
        danmuList: dbCmd.push([danmuItem])
    })
    console.log(res);
    //返回数据给客户端
    let data = await danmuCollection.where({
        videoid
    }).get()
    return data
};
