# 手写深拷贝
```js
function deepClone(o){
    if(typeof o ！== 'object' || o == null){
        return o
    }
    let n;
    if(Object.prototype.toString.call(o)==='[object Ojbect]'){
        n = {}
    }else{
        n = []
    }
    for(let key in o){
        if(o.hasOwnproperty(key)){
            n[key] = deepClone(o[key])
        }
    }
    return n;
}
```
